function minifyCSS(code) {
  var stripComments = document.getElementById('chk-css-comments').checked;
  var stripZeros = document.getElementById('chk-css-zeros').checked;
  var shortenColors = document.getElementById('chk-css-colors').checked;

  /* Strip wrapping  tags and restore them after minification */
  var openTag = '', closeTag = '';
  var styleMatch = code.match(/^(\s*]*>)([\s\S]*?)(<\/style>\s*)$/i);
  if (styleMatch) {
    openTag  = styleMatch[1];
    code     = styleMatch[2];
    closeTag = styleMatch[3];
  }

  var out = code;
  if (stripComments) out = out.replace(/\/\*[\s\S]*?\*\//g, '');
  out = out
    .replace(/\r\n|\r/g, '\n').replace(/\s*\{\s*/g, '{').replace(/\s*\}\s*/g, '}')
    .replace(/\s*:\s*/g, ':').replace(/\s*;\s*/g, ';').replace(/\s*,\s*/g, ',')
    .replace(/\n+/g, ' ').replace(/[ \t]+/g, ' ').trim();
  out = out.replace(/;}/g, '}');
  if (stripZeros) {
    out = out.replace(/\b0(px|em|rem|%|vh|vw|vmin|vmax|pt|pc|cm|mm|in|ex|ch)\b/g, '0');
    out = out.replace(/:0\.(\d)/g, ':.$1');
    out = out.replace(/ 0\.(\d)/g, ' .$1');
  }
  if (shortenColors) {
    out = out.replace(/#([0-9a-fA-F])\1([0-9a-fA-F])\2([0-9a-fA-F])\3\b/g, '#$1$2$3');
    out = out.replace(/([:,\s(])white\b/g, '$1#fff');
    out = out.replace(/([:,\s(])black\b/g, '$1#000');
  }

  return openTag ? openTag + '\n' + out + '\n' + closeTag.trim() : out;
}

function formatBytes(n) {
  if (n < 1024) return n + ' B';
  return (n / 1024).toFixed(1) + ' KB';
}

function applyResults(errors, warnings, doMinify) {
  var code = document.getElementById('input').value.trim();
  var inputPanel = document.getElementById('input-panel');
  var outputPanel = document.getElementById('output-panel');
  var outputEl = document.getElementById('output');

  /* Build annotation list for gutter + highlights */
  var annotations = [];
  for (var i = 0; i < errors.length; i++)
    annotations.push({ line: errors[i].line, msg: errors[i].msg, type: 'error' });
  for (var i = 0; i < warnings.length; i++)
    annotations.push({ line: warnings[i].line, msg: warnings[i].msg, type: 'warning' });

  applyAnnotations(annotations);

  if (errors.length) {
    inputPanel.className = 'editor-panel error';
    outputPanel.className = 'editor-panel';
    outputEl.value = '';
    var lineInfo = errors[0].line ? ' on line ' + errors[0].line : '';
    setStatus('invalid', ICONS.x, '' + (currentMode === 'css' ? 'CSS error' : 'Syntax error') + lineInfo + '. ' + errors.length + ' error' + (errors.length > 1 ? 's' : '') + ' highlighted below.');
    ['m-orig','m-min','m-saved','m-pct'].forEach(function(id) { document.getElementById(id).textContent = '—'; });
    return;
  }

  if (doMinify) {
    var minified = currentMode === 'css' ? minifyCSS(code) : minifyJS(code);
    outputEl.value = minified;
    var origBytes = new Blob([code]).size;
    var minBytes = new Blob([minified]).size;
    var saved = origBytes - minBytes;
    var pct = origBytes > 0 ? Math.round((saved / origBytes) * 100) : 0;
    document.getElementById('m-orig').textContent = formatBytes(origBytes);
    document.getElementById('m-min').textContent = formatBytes(minBytes);
    document.getElementById('m-saved').textContent = formatBytes(saved);
    document.getElementById('m-pct').textContent = pct + '%';
    inputPanel.className = 'editor-panel' + (warnings.length ? '' : ' valid');
    outputPanel.className = 'editor-panel' + (warnings.length ? '' : ' valid');
    var msg = 'Valid. Minified successfully — ' + pct + '% smaller.';
    if (warnings.length) {
      msg = 'Minified with ' + warnings.length + ' warning' + (warnings.length > 1 ? 's' : '') + '. Check highlighted lines.';
      setStatus('warning', ICONS.warn, msg);
    } else {
      setStatus('valid', ICONS.check, msg);
    }
  } else {
    inputPanel.className = 'editor-panel' + (warnings.length ? '' : ' valid');
    var msg = 'Syntax valid. No errors found.';
    if (warnings.length) {
      msg = 'Valid with ' + warnings.length + ' warning' + (warnings.length > 1 ? 's' : '') + '. Check highlighted lines.';
      setStatus('warning', ICONS.warn, msg);
    } else {
      setStatus('valid', ICONS.check, msg);
    }
  }
}

function runAll() {
  var code = document.getElementById('input').value.trim();
  var lang = currentMode === 'css' ? 'CSS' : 'JavaScript';
  if (!code) { setStatus('warning', ICONS.warn, 'No input — paste your ' + lang + ' first.'); return; }
  var result = currentMode === 'css' ? checkSyntaxCSS(code) : checkSyntaxJS(code);
  applyResults(result.errors, result.warnings, true);
}

function checkOnly() {
  var code = document.getElementById('input').value.trim();
  if (!code) { setStatus('warning', ICONS.warn, 'No input — paste your JavaScript first.'); return; }
  var result = currentMode === 'css' ? checkSyntaxCSS(code) : checkSyntaxJS(code);
  applyResults(result.errors, result.warnings, false);
}

function copyOutput() {
  var out = document.getElementById('output').value;
  if (!out) return;
  navigator.clipboard.writeText(out).then(function() {
    setStatus('valid', ICONS.clip, 'Copied to clipboard!');
    setTimeout(function() { setStatus('', ICONS.info, 'Ready.'); }, 2000);
  });
}

function clearAll() {
  document.getElementById('input').value = '';
  document.getElementById('output').value = '';
  document.getElementById('input-panel').className = 'editor-panel';
  document.getElementById('output-panel').className = 'editor-panel';
  document.getElementById('in-chars').textContent = '0 chars';
  ['m-orig','m-min','m-saved','m-pct'].forEach(function(id) { document.getElementById(id).textContent = '—'; });
  clearHighlights();
  var lang = currentMode === 'css' ? 'CSS' : 'JavaScript';
  setStatus('', ICONS.info, 'Paste ' + lang + ' above and click Minify & Check.');
}

function loadSample() {
  document.getElementById('input').value = currentMode === 'css' ? CSS_SAMPLE : JS_SAMPLE;
  updateInputCount();
  clearHighlights();
  document.getElementById('input-panel').className = 'editor-panel';
  setStatus('', ICONS.info, 'Sample loaded — click Minify & Check to run.');
}

function printSinglePage() {
  var sh = document.body.scrollHeight, sw = document.body.scrollWidth;
  var style = document.createElement('style');
  style.id = '_print_override';
  style.textContent = '@page { size: ' + sw + 'px ' + sh + 'px; margin: 0; } body { zoom: 1; }';
  document.head.appendChild(style);
  window.print();
  setTimeout(function() { var s = document.getElementById('_print_override'); if (s) s.remove(); }, 1000);
}

document.addEventListener('keydown', function(e) {
  if ((e.ctrlKey || e.metaKey) && e.key === 'Enter') { e.preventDefault(); runAll(); }
});
